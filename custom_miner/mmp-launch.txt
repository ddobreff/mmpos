#!/usr/bin/env bash
# --- DEFAULT CMD ARGUMENTS FOR CUSTOM MINERS ---
# ./mmp-launch.sh --coin %coin% --pool %pool_server%:%pool_port% --user %user% --password %password% --api-port %api_port%
# --- END DEFAULT CMD ARGUMENTS ---
EXEC="./my-custom-miner"
CONF_FILE="mmp-external.conf"

# ---------------- MINER CAPABILITY CONFIG ----------------
# 1. Coin/Algo Mapping
declare -A COIN_MAP
COIN_MAP["XNT"]="xnt-algo-name" 
COIN_MAP["BTC"]="sha256"
COIN_MAP["ETHW"]="etchash"

# 2. Wallet/Worker Mapping Setup
MINER_USES_SPLIT=true       # Change to false if miner wants a single unified string
WAL_FLAG="--wallet"          # Flag name for wallet/address (e.g., --wallet, --address, -u)
WRK_FLAG="--worker"          # Flag name for worker (e.g., --worker, --rig, -p)
USR_FLAG="--user"            # Fallback flag name if MINER_USES_SPLIT=false


# ---------------- PARSE GENERIC ARGS ----------------
ARGS=("$@")
FINAL_ARGS=()

POOL_VAL=""
USER_VAL=""
PASS_VAL=""
COIN_VAL=""
API_PORT=4444

i=0
while [[ $i -lt ${#ARGS[@]} ]]; do
    case "${ARGS[$i]}" in
        --pool)
            POOL_VAL="${ARGS[$((i+1))]}"
            ((i+=2))
            ;;
        --user)
            USER_VAL="${ARGS[$((i+1))]}"
            ((i+=2))
            ;;
        --password)
            PASS_VAL="${ARGS[$((i+1))]}"
            ((i+=2))
            ;;
        --coin)
            COIN_VAL="${ARGS[$((i+1))]}"
            ((i+=2))
            ;;
        --api-port)
            API_PORT="${ARGS[$((i+1))]}"
            ((i+=2))
            ;;
        *)
            FINAL_ARGS+=("${ARGS[$i]}")
            ((i+=1))
            ;;
    esac
done

# ---------------- DEFAULTS & CONFIG ----------------
[[ -z "$API_PORT" || "$API_PORT" == "0" ]] && API_PORT=4444

# Persist API Port
if grep -q '^CUSTOM_API_PORT=' "$CONF_FILE" 2>/dev/null; then
    sed -i "s/^CUSTOM_API_PORT=.*/CUSTOM_API_PORT=$API_PORT/" "$CONF_FILE"
else
    echo "CUSTOM_API_PORT=$API_PORT" >> "$CONF_FILE"
fi

# ---------------- SPLIT WALLET AND WORKER LOGIC ----------------
WALLET_ADDRESS=""
WALLET_WORKER=""

if [[ -n "$USER_VAL" ]]; then
    if [[ "$USER_VAL" == *.* ]]; then
        WALLET_ADDRESS="${USER_VAL%%.*}"
        WALLET_WORKER="${USER_VAL#*.}"
    else
        WALLET_ADDRESS="$USER_VAL"
        WALLET_WORKER="$(hostname)"
    fi
fi

# ---------------- BUILD MINER CMD ----------------
CMD=( "$EXEC" )

# 1. Handle Coin/Algo Mapping
if [[ -n "$COIN_VAL" ]]; then
    if [[ -n "${COIN_MAP[$COIN_VAL]}" ]]; then
        CMD+=( --algo "${COIN_MAP[$COIN_VAL]}" ) 
    else
        echo "Warning: Coin '$COIN_VAL' has no valid mapper defined. Skipping coin/algo argument." >&2
    fi
fi

# 2. Handle Pool
[[ -n "$POOL_VAL" ]] && CMD+=( --pool "$POOL_VAL" )

# 3. Handle Wallet / Worker formatting based on config
if [[ "$MINER_USES_SPLIT" = true ]]; then
    [[ -n "$WALLET_ADDRESS" ]] && CMD+=( "$WAL_FLAG" "$WALLET_ADDRESS" )
    [[ -n "$WALLET_WORKER" ]]  && CMD+=( "$WRK_FLAG" "$WALLET_WORKER" )
else
    [[ -n "$USER_VAL" ]]       && CMD+=( "$USR_FLAG" "$USER_VAL" )
fi

# 4. Handle Password & API
[[ -n "$PASS_VAL" ]] && CMD+=( --pass "$PASS_VAL" )
CMD+=( --api-port "$API_PORT" )

# 5. Append any remaining unknown arguments as-is
CMD+=( "${FINAL_ARGS[@]}" )

# ---------------- RUN ----------------
echo "Running: ${CMD[*]}"
exec "${CMD[@]}"
